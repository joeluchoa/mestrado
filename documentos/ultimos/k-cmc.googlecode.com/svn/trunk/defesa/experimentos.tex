\subsection*{Experimentos}

\begin{frame}
\onslide<1->{
\begin{block}{Variáveis}
\begin{itemize}
\item $n$: número de vértices
\item $m$: número de arestas
\item $d$: densidade=$m/\binom{n}{2}$
\item $k$: quantidade de menores caminhos
\end{itemize}
\end{block}
}
\onslide<2->{
\begin{block}{Metodologia}
\begin{itemize}
\item Grafos simétricos conexos com custos positivos 
\item 5 pontas iniciais e finais
\item Um grafo por densidade
\end{itemize}
\end{block}
}
\onslide<3->{
\begin{block}{Consumo de tempo}
\begin{itemize}
\item \KIM: $\Theta(kT(n,m))=\Theta(km\log n)$ usando-se \Dijkstra
\item \KIM: $\Theta(km\log n)=\Theta(kdn^2\log n)$
\end{itemize}
\end{block}
}
\end{frame}

\begin{frame}{Principais rotinas}
\begin{block}{Rotinas}
\begin{itemize}
\item Problema do desvio mínimo, rotina \SEP 
\item Problema \PCM, algoritmo Dijkstra
\end{itemize}
\end{block}

\begin{block}{Consumo de tempo}
\begin{itemize}
\item \SEP : $\Theta(m+n)$ 
\item \Dijkstra{} com "min-heap": $\Theta(m\log n)$
\end{itemize}
\end{block}
\end{frame}
